async function saveDiary() { const menuType = document.getElementById('menu-select').value; let dist = 0, detail = "", ttTime = ""; if (menuType === 'ジョグ') { dist = parseFloat(document.getElementById('jog-dist').value); detail = document.getElementById('jog-res').innerText; } else if (menuType === 'インターバル' || menuType === 'レスト') { const d = parseFloat(document.getElementById('int-dist').value) || 0; const c = parseFloat(document.getElementById('int-count').value) || 0; dist = d * c; detail = `${d}k×${c}`; } else if (menuType === 'タイムトライアル') { dist = parseFloat(document.getElementById('tt-dist-km').value) || 0; ttTime = `${document.getElementById('tt-min').value}:${document.getElementById('tt-sec').value}.${document.getElementById('tt-msec').value}`; detail = `${dist}k TT [${ttTime}]`; } else if (menuType === 'その他') { dist = parseFloat(document.getElementById('other-dist').value) || 0; detail = document.getElementById('other-content').value; } if (isNaN(dist) && menuType !== 'レスト') return alert("距離を入力してください"); const btn = document.getElementById('save-btn'); btn.innerText = "保存中..."; btn.disabled = true; let w = "☁️", t = "--°C"; try { // --- 時間取得 --- const startTime = document.getElementById('start-time').value || "12:00"; const [hour, minute] = startTime.split(":"); const targetDateTime = `${selectedDateStr}T${hour}:${minute}`; // --- 日付判定（未来は天気取得しない） --- const today = new Date().toISOString().split('T')[0]; if (selectedDateStr <= today) { // --- 位置取得 --- let lat, lon; const selectedCity = document.getElementById('city-select').value; if (selectedCity) { [lat, lon] = selectedCity.split(','); } else { const pos = await new Promise((res, rej) => navigator.geolocation.getCurrentPosition(res, rej) ); lat = pos.coords.latitude; lon = pos.coords.longitude; } // --- 過去天気取得 --- const res = await fetch( `https://archive-api.open-meteo.com/v1/archive?latitude=${lat}&longitude=${lon}&start_date=${selectedDateStr}&end_date=${selectedDateStr}&hourly=temperature_2m,weathercode` ); const data = await res.json(); const index = data.hourly.time.findIndex(t => t === targetDateTime); if (index !== -1) { const tempVal = data.hourly.temperature_2m[index]; t = Math.round(tempVal) + "°C"; const code = data.hourly.weathercode[index]; w = convertWeatherCode(code); } } } catch (e) { console.log("天気取得失敗", e); } const diary = { date: selectedDateStr, type: menuType, dist: dist.toFixed(2), detail: detail, weather: w, temp: t, ttTime: ttTime }; try { await db.collection('users') .doc(auth.currentUser.uid) .collection('diaries') .doc(selectedDateStr) .set(diary); closeModal(); loadCloudData(); } catch (e) { alert("保存失敗: " + e.message); } btn.innerText = "日誌を保存（天気取得）"; btn.disabled = false; }
