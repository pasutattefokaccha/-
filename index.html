async function handleAuth() { const btn = document.getElementById('auth-btn'); const email = document.getElementById('email').value.trim(); const password = document.getElementById('password').value; const errDiv = document.getElementById('auth-error'); if (!errDiv) return; // エラー表示エリアがない場合は中断 errDiv.style.display = 'none'; if (!email || !password || password.length < 6) { errDiv.innerText = "メールと6文字以上のパスワードを入力してください"; errDiv.style.display = 'block'; return; } btn.innerText = "通信中..."; btn.disabled = true; try { // 1. まずログインを試みる await auth.signInWithEmailAndPassword(email, password); } catch (error) { // 2. ユーザーが存在しない等のエラー時に新規登録を試みる if (error.code === 'auth/user-not-found' || error.code === 'auth/invalid-login-credentials' || error.code === 'auth/user-disabled') { try { await auth.createUserWithEmailAndPassword(email, password); } catch (regErr) { errDiv.innerText = "エラー: " + regErr.message; errDiv.style.display = 'block'; } } else { errDiv.innerText = "エラー: " + error.message; errDiv.style.display = 'block'; } } finally { btn.innerText = "ログイン / 新規登録"; btn.disabled = false; } } // HTML側のボタンが以下のように「onclick="handleAuth()"」となっているか確認してください //
